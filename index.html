window.spinWheel = async function() { const email = document.getElementById('userEmail').innerText; const spinBtn = document.getElementById('spinBtn'); const spinMsg = document.getElementById('spinMessage'); try { const { data: userData, error: fetchError } = await supabase .from('users') .select('last_spin, wallet_ads') .eq('email', email) .single(); if (fetchError) { alert('خطا در دریافت اطلاعات'); return; } const lastSpin = userData.last_spin ? new Date(userData.last_spin) : null; const now = new Date(); const hoursDiff = lastSpin ? (now - lastSpin) / (1000 * 60 * 60) : 999; if (hoursDiff < 48) { const remaining = Math.ceil(48 - hoursDiff); spinMsg.innerHTML = `⏳ هنوز ${remaining} ساعت مونده تا چرخش بعدی`; return; } spinBtn.disabled = true; spinBtn.innerText = '🌀 در حال چرخش...'; spinMsg.innerHTML = ''; const prize = Math.floor(Math.random() * 901) + 100; const newBalance = (userData.wallet_ads || 0) + prize; const { error: updateError } = await supabase .from('users') .update({ wallet_ads: newBalance, last_spin: now.toISOString() }) .eq('email', email); if (updateError) { alert('خطا در ثبت چرخش'); spinBtn.disabled = false; spinBtn.innerText = '🔄 چرخش (هر ۴۸ ساعت یک بار)'; return; } document.getElementById('walletAds').innerText = newBalance; spinMsg.innerHTML = `🎉 تبریک! ${prize} تومان به کیف پول تبلیغ اضافه شد!`; spinBtn.disabled = false; spinBtn.innerText = '🔄 چرخش (هر ۴۸ ساعت یک بار)'; } catch (err) { alert('خطا: ' + err.message); spinBtn.disabled = false; spinBtn.innerText = '🔄 چرخش (هر ۴۸ ساعت یک بار)'; } };
